<h2>XMLHttpRequest Örnek Uygulama</h2>
<h3>Aşağıdaki alana bir isim yazmaya başlayın:</h3>

<p>Öneriler: <span id="txtHint"></span></p> 
<p>İsim: <input type="text" id="txt1" onkeyup="showHint(this.value)"></p>

<script>
function showCustomer(str) {
  if (str == "") {
    document.getElementById("txtHint").innerHTML = "Müşteri seçilmedi";
    return;
  }
  const xhttp = new XMLHttpRequest();
  xhttp.onload = function() {
    document.getElementById("txtHint").innerHTML = this.responseText;
  }
  xhttp.open("GET", "musteriler.php?id="+str);
  xhttp.send();
}
</script>
